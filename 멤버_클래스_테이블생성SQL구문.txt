CREATE TABLE MEMBER
(
    `MEM_ID`      VARCHAR(45)    NOT NULL    COMMENT '아이디', 
    `MEM_PW`      VARCHAR(45)    NOT NULL    COMMENT '비밀번호', 
    `MEM_NAME`    VARCHAR(45)    NOT NULL    COMMENT '이름', 
    `MEM_PHONE`   VARCHAR(45)    NULL        COMMENT '전화번호', 
    `MEM_EMAIL`   VARCHAR(45)    NOT NULL    COMMENT '이메일', 
    `MEM_CODE`    VARCHAR(45)    NOT NULL    COMMENT '회원번호', 
    `MEM_ACCESS`  TINYINT        NOT NULL    COMMENT '권한', 
    `MEM_DATE`    DATE           NULL        COMMENT '가입일', 
    `MEM_STATUS`  BLOB           NULL        COMMENT '탈퇴여부', 
    PRIMARY KEY (MEM_CODE)
);

ALTER TABLE MEMBER
    ADD CONSTRAINT FK_MEMBER_MEM_CODE_CLASS_MEM_CODE FOREIGN KEY (MEM_CODE)
        REFERENCES CLASS (MEM_CODE) ON DELETE RESTRICT ON UPDATE RESTRICT;

CREATE TABLE CLASS
(
    `CLS_CODE`      INT            NOT NULL    COMMENT '강의코드', 
    `CLS_CATEGORY`  VARCHAR(45)    NOT NULL    COMMENT '강의분류', 
    `CLS_NAME`      VARCHAR(45)    NOT NULL    COMMENT '강의명', 
    `MEM_CODE`      VARCHAR(45)    NOT NULL    COMMENT '회원번호', 
    `CLS_PRICE`     VARCHAR(45)    NOT NULL    COMMENT '가격', 
    `CLS_TIME`      TIME           NOT NULL    COMMENT '재생시간', 
    `CLS_PERIOD`    DATE           NOT NULL    COMMENT '강의기간', 
    PRIMARY KEY (CLS_CODE)
);
